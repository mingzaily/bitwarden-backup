```sql
-- SQLite
ALTER TABLE server_configs ALTER COLUMN enabled SET DEFAULT TRUE;
ALTER TABLE backup_destinations ALTER COLUMN enabled SET DEFAULT TRUE;
```

```vue
<template>

<script setup>
// 添加方法
const isOfficialServer = (server) => {
  const officialUrls = ['https://vault.bitwarden.com', 'https://vault.bitwarden.eu']
  return officialUrls.includes(server.server_url)
}

const toggleServer = async (id, enabled) => {
  try {
    await serversApi.update(id, { enabled })
    toast.success(enabled ? '已启用' : '已禁用')
    loadServers()
  } catch (error) {
    console.error('Failed to toggle server:', error)
    toast.error('操作失败')
  }
}
</script>
```
